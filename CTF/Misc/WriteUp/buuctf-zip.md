set zipfile_list (ls *.zip)
set num_suffix (seq 0 67)
set tmpfile (mktemp)

for i in $num_suffix
	echo "Start crack out$i.zip"
	set tmp (zipinfo -v ./out$i.zip | grep CRC | tr -d ' ' | cut -d: -f2)
	set result (crc32 reverse 0x$tmp | head -n 1 | cut -d'{' -f2 | cut -d'}' -f 1)
	echo $result >> $tmpfile
end
echo (set_color green)"The result save to $tmpfile"(set_color normal)
